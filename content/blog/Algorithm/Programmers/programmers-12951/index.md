```js
function solution(s) {
  let answer = "";
  answer = s
    .toLowerCase()
    .split(" ")
    .map((word) => {
      if (word === "") return "";
      return word.at(0).toUpperCase() + word.slice(1);
    });
  answer = answer.join(" ");
  return answer;
}
```
